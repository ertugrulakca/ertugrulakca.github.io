### Objects

```js
var myObj = { foo: 'bar' }

console.log(myObj); // { foo: 'bar' }
console.log(myObj.foo); // bar
```


### Arrays

```js
const myArr = [1, 2, 3, 4, 5];

myArr.forEach(function (value) {
    console.log(value);
})

console.log(myArr); // [1, 2, 3, 4, 5]

/*
output: 
1
2
3
4
5
*/
```

### Exception Handling

```js
const num1 = 1;
const num2 = 0;

try {
    if (num2 === 0) {
        throw Error("Division by zero error");
    }

    const num3 = num1 / num2;
    console.log(num3);
} catch (error) {
    console.log("Error Catched: " + error.message);
}
```
