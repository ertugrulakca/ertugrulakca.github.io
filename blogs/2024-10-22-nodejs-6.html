Functions

function sum(a, b = 0) {
    return a + b;
}

console.log(sum(5, 3)); // 8
console.log(sum(5)); // 5

Callback Functions

function run(text, callback) {
    console.log(text);
    callback();
} 

run('Hello World', function () {
    console.log('Callback Function')
})

"setTimeout" function

setTimeout(function () {
    console.log('Wrote after 5 seconds');
}, 5000)

"setInterval" function

const myInterval = setInterval(function () {
    console.log('Writes every 2 seconds forever');
}, 2000);

setTimeout(function () {
    clearInterval(myInterval);
}, 7000);

For Loop

for (var i = 0; i < 5; i++) {
    console.log(i)
}

/*
output:
0
1
2
3
4
*/

While Loop

var i = 0;

while (i < 5) {
    console.log(i++);
}

/*
output:
0
1
2
3
4
*/

Do-While Loop

var i = 0;

do {
    console.log(i++);  
} while (i < 5);

/*
output:
0
1
2
3
4
*/
